-- usersテーブル作成
CREATE TABLE users (
    id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR2(100),
    email VARCHAR2(200),
    role VARCHAR2(5) DEFAULT 'USER' CHECK (role IN ('USER', 'ADMIN', 'GUEST')),
    created_at DATE DEFAULT SYSDATE
);
